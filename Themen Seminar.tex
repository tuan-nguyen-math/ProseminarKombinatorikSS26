\section{Themen}
Die vorzutragenden Themen basieren auf den Büchern \cite{aigner2006diskrete,hougardy2018algorithmische}:
\begin{enumerate}
    \item Primzahltest und Berechenbarkeit (\cite[Kapitel 1]{hougardy2018algorithmische})
    \item Darstellungen ganze Zahlen (\cite[Kapitel 2]{hougardy2018algorithmische})
    \item Rechnen mit ganzen Zahlen (\cite[Kapitel 3]{hougardy2018algorithmische})
    \item Approximative Darstellung reeller Zahlen (\cite[Kapitel 4]{hougardy2018algorithmische})
    \item Rechnen mit Fehlern (\cite[Kapitel 5]{hougardy2018algorithmische})
    \item Graphen (\cite[Kapitel 6]{hougardy2018algorithmische})
    \item Einfache Graphenalgorithmen (\cite[Kapitel 7]{hougardy2018algorithmische})
    \item Sortieralgorithmen (\cite[Kapitel 8]{hougardy2018algorithmische})
    \item Optimale Bäume und Wege ((\cite[Kapitel 9]{hougardy2018algorithmische}))    
   % \item Gauss-Elimination (\cite[Kapitel 11]{hougardy2018algorithmische})
    \item Elementare Zählprinzipien und Permutationen (\cite[Abschnitte 1.1--1.3]{aigner2006diskrete})
    \item Rekursion und Existenzaussagen (\cite[Abschnitte 1.4--1.6]{aigner2006diskrete})
    \item Summation -- Teil 1 (\cite[Abschnitte 2.1--2.2]{aigner2006diskrete})
    \item Summation -- Teil 2 (\cite[Abschnitte 2.3--2.4]{aigner2006diskrete})
    \item Erzeugende Funktionen (\cite[Kapitel 3]{aigner2006diskrete})
    \item Gauss-Elimination (\cite[Kapitel 11]{hougardy2018algorithmische})
\end{enumerate}

\end{enumerate}
